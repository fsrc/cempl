cempl

Javascript functional templating engine, best in use with Coffee-Script.

It’s more of a generator. It doesn’t parse the template. It executes it. Example:

html = require('cempl')

standardHeader = (title) ->
  @head(->
    @meta('charset':'utf-8', -> )
    @meta('X-UA-Compatible':'IE=edge', -> )
    @meta('Content-Language':'en', -> )
    @title(title))

raw = html(->
  @apply(standardHeader, 'My Title')
  @body(->
    @div(->
      @text("Hello")))).eval()

console.log(raw)

Same example in Javascript. It's generated by Coffee-Script so it might become more beautiful if handwritten.

// Generated by CoffeeScript 1.10.0
var html, raw, standardHeader;

html = require('../index');

standardHeader = function(title) {
  return this.head(function() {
    this.meta({
      'charset': 'utf-8'
    }, function() {});
    this.meta({
      'X-UA-Compatible': 'IE=edge'
    }, function() {});
    this.meta({
      'Content-Language': 'en'
    }, function() {});
    return this.title(title);
  });
};

raw = html(function() {
  this.apply(standardHeader, 'My Title');
  return this.body(function() {
    return this.div(function() {
      return this.text("Hello");
    });
  });
})["eval"]();

console.log(raw);
